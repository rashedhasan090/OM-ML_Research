sig Record {
    a: Int,
    b: Int,
    c: Int
}

fun join[P, Q: Record]: set Record {
    { r: Record |
        some p: P, q: Q |
        p.b = q.b and r.a = p.a and r.b = p.b and r.c = q.c
    }
}

pred JoinVerification[P, Q: Record] {
    all r: Record |
        r in join[P, Q] <=> (
            some p: P, q: Q |
            p.b = q.b and r.a = p.a and r.b = p.b and r.c = q.c
        )
}

run JoinVerification for 4 but 4 Record
